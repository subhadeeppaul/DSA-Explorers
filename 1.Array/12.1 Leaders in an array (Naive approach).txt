#include<bits/stdc++.h>
using namespace std;

void leaders(int arr[], int n)
{
    for(int i=0; i<n; i++)
    {
        bool flag=false;
        for(int j=i+1;j<n;j++)
        {
            if(arr[i]<=arr[j])
            {
                flag=true;
                break;
            }
        }
        
        if (flag==false)
        {
            cout<<arr[i]<< " ";
        }
    }
}

int main()
{
    int n;
    cin>>n;
    
    int arr[n];
    
    for(int i=0;i<n;i++)
    {
        cin>>arr[i];
    }
    
    leaders(arr,n);
 }


// Time Complexity: O(n^2)
