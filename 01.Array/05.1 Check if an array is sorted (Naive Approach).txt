#include<iostream>
using namespace std;


int isSorted(int a[], int n)
{
    for(int i=0;i<n;i++)
    {
        for(int j=i+1;j<n;j++)
        {
            if(a[j]<a[i])
                return false;
        }
 
    }
return true;


}

int main()
{
    int n;
    cin>>n;

    int a[n];

    for(int i=0;i<n;i++)
    {
        cin>>a[i];

    }

    if(isSorted(a,n)){
        cout<<" Yes";
    }
    else{
        cout<<"No";
    }

}
