#include <iostream>
using namespace std;

 void moveToEnd(int arr[], int n)
 {


     for(int i=0;i<n;i++)
     {
         if(arr[i]==0)
         {
             for(int j=i+1;j<n;j++)
                if(arr[j]!=0)
                {
                   swap(arr[i],arr[j]);
                   break;

                }

         }


     }

}

int main()
{

    int n;
    cin>>n;

    int arr[n];


    for(int i=0;i<n;i++)
    {
        cin>>arr[i];
    }

    moveToEnd(arr,n);

    cout<<"Array after pushing all zeros to end of array :\n";
    for(int i=0;i<n;i++)
        cout<<arr[i]<<" ";

    return 0;

}
