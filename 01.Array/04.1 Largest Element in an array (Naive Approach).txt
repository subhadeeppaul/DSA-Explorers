#include<iostream>
using namespace std;


int getLargest(int a[], int n)
{
    for(int i=0;i<n;i++)
    {
        bool flag=true;

        for(int j=0;j<n;j++)
        {
            if(a[j]>a[i])
            {
                flag=false;
                break;
            }


        }

        if(flag==true)
        {

            return i;
        }

    }

    return -1;

}

int main()
{

    int n;
    cin>>n;

    int a[n];

    for(int i=0;i<n;i++)
    {
        cin>>a[i];

    }

cout<<getLargest(a,n);
return 0;
}
