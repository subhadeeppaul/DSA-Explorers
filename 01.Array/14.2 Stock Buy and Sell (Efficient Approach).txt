#include<bits/stdc++.h>
using namespace std;

int maxProfit(int price[],int n)
{
    
  int profit=0;
  
  for(int i=1;i<n;i++)
  {
      if(price[i]>price[i-1])
      {
          profit += (price[i]-price[i-1]);
      }
  }
  
  return profit;
}

int main()
{
    int price;
    cin>>price;
    
    int arr[price];
    
    for(int i=0;i<price;i++)
    {
        cin>>arr[i];
    }
    cout<<maxProfit(arr,price)<<" ";
    
}
